    lw 0 1 A        //Register@1 = 7    		//Half Adder
	lw 0 2 B		//Register@2 = 5
	nand 1 2 3		//Register@3 = Register@1(7) & Register@2(5) 
	nand 1 3 4		//Register@4 = Register@1(7) & Register@3
	nand 2 3 5		//Register@5 = Register@2(5) & Register@3
	nand 4 5 4		//Register@4 = Register@5 & Register@4
	nand 3 3 3		//Register@3 = Register@3 & Register@3
done	halt		//end of program
	A .fill 7		// A = 7
	B .fill 5		// B = 5
